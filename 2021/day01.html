func countIncrease(s: seq[int]): int =
  for i in 1 .. s.high:
    if s[i] > s[i - 1]:
      inc result

func window(s: seq[int]; size = 3): seq[int] =
  assert s.len >= size
  var sum = s[0 ..< size].sum
  result.add sum
  for i in size .. s.high:
    sum = sum - s[i - size] + s[i]
    result.add sum

echo input[0 ..< 10]
echo input[0 ..< 10].window
