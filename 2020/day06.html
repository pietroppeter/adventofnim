type
  Answer = char
  AnsSet = set[char]
proc solve(text: string): (int, int) =
  var text = text & "\n\n"
  var
    groupAny: AnsSet
    groupEvery: AnsSet
    groupNew = true
    answers: AnsSet
    answer: Answer
    b: char
  for c in text:
    if c.isLowerAscii:
      answer = c.Answer
      answers.incl(answer)
    elif b == '\n':
      result[0] += groupAny.len
      result[1] += groupEvery.len
      groupAny = {}.AnsSet
      groupEvery = {}.AnsSet
      groupNew = true
    elif c == '\n':
      groupAny.incl(answers)
      if groupNew:
        groupEvery = answers
        groupNew = false
      else:
        groupEvery = groupEvery * answers
      answers = {}.AnsSet
    else:
      echo "ERROR parsing " & c.repr
    b = c
