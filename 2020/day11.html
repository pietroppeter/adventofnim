proc parse(text: string): SeatGrid =
  for i, c in text:
    case c
    of '.':
      result.data.add sgFloor
    of 'L':
      result.data.add sgEmpty
    of '\n':
      if result.ncols == 0:
        result.ncols = result.data.len
      elif result.data.len mod result.ncols != 0:
        echo "ERROR expect same number of columns ", i, " ", c

    if i >= g.ncols:
