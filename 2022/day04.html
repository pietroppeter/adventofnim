func parse(text: string): seq[SectionPairs] =
  var sp: SectionPairs
  for line in text.splitLines:
    if not scanf(line, "$i-$i,$i-$i", sp[0].a, sp[0].b, sp[1].a, sp[1].b):
      debugEcho "ERROR, line: '", line, "'"
    else:
      result.add sp
